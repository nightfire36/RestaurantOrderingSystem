create table Country (CountryID INT PRIMARY KEY, Name VARCHAR(20));

create table City (CityID INT PRIMARY KEY, CountryID INT, Name VARCHAR(20), Population INT, FOREIGN KEY(CountryID) REFERENCES Country(CountryID));

create table Building (BuildingID INT PRIMARY KEY, CityID INT, Name VARCHAR(20), Floors INT, FOREIGN KEY (CityID) REFERENCES City(CityID));
